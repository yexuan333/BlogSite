int FindSpan(n,p,u,U){
    if(u == U[n+1]) return n
    low = p;
    hight = n+1;
    mid = (low+hight)/2;
    while(u<U[mid] || u>U[mid+1]){
        if(u<U[mid]) hight = mid;
        else low = mid;
    }
    return (mid);
}

BasisFuns(i,u,p,U,N){
    N[0]=1.0
    for(j=1;j<=p;j++>){
        left[j]=u-U[i+1-j];
        right[j]=U[i+j]-u;
        saved = 0.0;      
        for(r=0;r<j;r++){
            temp=N[r]/(right[r+1]+left[j-r]);
            N[r]=saved+right[r+1]*temp;
            saved=left[j-r]*temp;
        }
        N[j]=saved;
    }
}
